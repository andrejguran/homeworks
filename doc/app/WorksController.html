def create
  @work = Work.new(params[:work])
  @work.task_id = params[:task_id]
  @work.user_id = current_user.id

  respond_to do |format|
    if @work.save

      if @work.task.language == "java"
        require "FileUtils"
        new_file_path = File.join(Rails.root, "homework/src", current_user.uco.to_s, @work.id.to_s, @work.task.package, @work.homework_file_name)
        FileUtils.mkdir_p(File.dirname(new_file_path))
        FileUtils.cp @work.homework.path, new_file_path

        classes_path = File.join(Rails.root, "homework/classes", current_user.uco.to_s, @work.id.to_s)
        FileUtils.mkdir_p(classes_path)
        work_file_path = new_file_path
        test_file_path = File.join(Rails.root, "homework/src/tests", @work.task.package, @work.task.task_file_file_name)
        command_compile_src_and_test = "javac -d " + classes_path +" " + work_file_path + " " + test_file_path
        response_compile_src_and_test = %x#{command_compile_src_and_test}`

        include_class_path = "-cp " + classes_path + ";" + Rails.root.to_s + "/lib/assets/junit.jar;" + Rails.root.to_s + "/lib/assets/hamcrest-core.jar "
        command_run_test = "java " + include_class_path  + "org.junit.runner.JUnitCore " + @work.task.package + "." + @work.task.task_file_file_name[0..-6]
        response_run_test = %x#{command_run_test}`

        @work.message = response_run_test

        if @work.message.include? "FAILURES!"
          @work.status = "fail"
        elsif @work.message.include? "OK"
          @work.status = "success"  
        end

        @work.save
      elsif @work.task.language == "xslt" || @work.task.language == "xquery"
        new_file_path = File.join(Rails.root, "homework/src", current_user.uco.to_s, @work.id.to_s, @work.task.package, @work.homework_file_name)
        FileUtils.mkdir_p(File.dirname(new_file_path))
        FileUtils.cp @work.homework.path, new_file_path

        
      end    

      format.html { redirect_to @work, notice: 'Work was successfully created.' }
      format.json { render json: @work, status: :created, location: @work }
      format.xml { render xml: @work, status: :created, location: @work }
    else
      format.html { render action: "new" }
      format.xml { render xml: @work.errors, status: :unprocessable_entity }
    end
  end
end

def generated
  @work = Work.find(params[:id])
  if params[:language] == "xslt"
    xsl_file = File.join(Rails.root, "homework/src", @work.user.uco.to_s, @work.id.to_s, @work.task.package, @work.homework_file_name)
    xml_file = File.join(Rails.root, "homework/src/tests", @work.task.package, @work.task.task_file_file_name)
    doc = Nokogiri::XML(File.read(xml_file))
    xslt = Nokogiri::XSLT(File.read(xsl_file))

    transform = xslt.transform(doc)

    render :text => transform
  elsif params[:language] == "xquery"
    xq_file = File.join(Rails.root, "homework/src", current_user.uco.to_s, @work.id.to_s, @work.task.package, @work.homework_file_name)
    command = "java -cp " + Rails.root.to_s + "/lib/assets/saxon9he.jar net.sf.saxon.Query " + xq_file
    response = %x#{command}`

    render :text => CodeRay.scan(response, :xml).div
  end
end

def homework
  @work = Work.find(params[:id])
  file_name = File.join(Rails.root, "homework/src", current_user.uco.to_s, @work.id.to_s, @work.task.package, @work.homework_file_name)
  file = File.open(file_name, "r")
  data = file.read
  file.close

  if @work.task.language == "java"
    render text:  CodeRay.scan(data, :java).div
  elsif @work.task.language == "xquery"
    render text:  CodeRay.scan(data, :php).div      
  else
    render text:  CodeRay.scan(data, :xml).div
  end
end

def update
  @work = Work.find(params[:id])

  respond_to do |format|
    if @work.update_attributes(params[:work])
      format.html { redirect_to work_url(@work), notice: 'Work was successfully updated.' }
      format.json { head :no_content }
      format.xml { head :no_content }
    else
      format.html { render action: "edit" }
      format.json { render json: @work.errors, status: :unprocessable_entity }
      format.xml { render xml: @work.errors, status: :unprocessable_entity }
    end
  end
end
